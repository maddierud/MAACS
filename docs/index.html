We will first consider the relationship between FEV1 and age. In general, it is expected that as children get older (and hence, larger in size), their FEV1 values should get higher.

Consider the statement FEV1 values in children are higher in older children relative to younger children.

Write a function in R that takes the dataset object as an argument and returns TRUE if the statement above is true for the dataset and FALSE otherwise.

# Check if older kids have higher FEV1 than younger kids
check_fev1_age <- function(data) {
  
  # Keep only age and fev1, drop missing values
  df <- data %>% select(age, fev1) %>% na.omit()
  
  # Find cutoffs for youngest and oldest thirds
  qs <- quantile(df$age, probs = c(1/3, 2/3))
  
  # Split into younger and older groups
  younger <- df %>% filter(age <= qs[1])
  older   <- df %>% filter(age >= qs[2])
  
  # Return TRUE if older mean FEV1 > younger mean FEV1
  mean(older$fev1) > mean(younger$fev1)
}

premise1_boxplot <- function(data) {

  qs <- quantile(data$age, probs = c(1/3, 2/3))
  data$age_group <- cut(data$age,
                        breaks = c(-Inf, qs[1], qs[2], Inf),
                        labels = c("Youngest", "Middle", "Oldest"))
  
  # Actual boxplot
  p_actual <- ggplot(data, aes(x = age_group, y = fev1, fill = age_group)) +
    geom_boxplot() +
    labs(title = "FEV1 by Age Group (Actual Data)",
         x = "Age Group", y = "FEV1 (liters)")

  # Hypothetical version via simulating clear increasing pattern
  set.seed(67)

premise3_slope <- function(data) {

  fit <- lm(fev1 ~ age, data = data)
  
  # Extract slope for age
  slope <- coef(fit)["age"]
  
  # Get 95% confidence interval for slope
  ci <- confint(fit)["age", ]
  
  # Return TRUE if slope > 0 and CI excludes 0
  (slope > 0) & (ci[1] > 0)
}
